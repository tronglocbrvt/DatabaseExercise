SELECT BM.MABM, BM.TENBM, COUNT(DISTINCT TG.MAGV) AS SLGV
FROM BOMON BM, THAMGIADT TG, DETAI DT, GIAOVIEN GV
WHERE BM.TRUONGBM = DT.GVCNDT AND DT.MADT = TG.MADT AND GV.MABM = BM.MABM AND GV.MAGV = TG.MAGV
GROUP BY BM.MABM, BM.TENBM

--CÂU 02
SELECT GV1.MAGV AS MAGV1, GV2.MAGV AS MAGV2
FROM GIAOVIEN GV1, GIAOVIEN GV2
WHERE GV1.NGSINH = GV2.NGSINH AND GV1.MAGV < GV2.MAGV
