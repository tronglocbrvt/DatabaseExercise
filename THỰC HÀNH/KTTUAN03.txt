--CÂU 01:
SELECT DISTINCT DT.MADT, DT.TENDT
FROM DETAI DT, GIAOVIEN GV_TTH, BOMON BM_THH, THAMGIADT TG, GIAOVIEN GV, BOMON BM_CNPM
WHERE GV_TTH.HOTEN = N'Trần Trà Hương' AND GV_TTH.MABM = BM_THH.MABM AND TG.MAGV = GV.MAGV 
	AND DT.GVCNDT = BM_CNPM.TRUONGBM AND BM_CNPM.MABM = 'CNPM' AND TG.MADT = DT.MADT

--CÂU 02:
SELECT DISTINCT CHUDE.*
FROM CHUDE, GIAOVIEN, THAMGIADT, DETAI
WHERE (GIAOVIEN.MABM = 'HTTT' OR GIAOVIEN.MABM = 'CNPM') AND (GIAOVIEN.MAGV = THAMGIADT.MAGV) 
	AND (THAMGIADT.MADT = DETAI.MADT) AND (DETAI.MACD = CHUDE.MACD)
